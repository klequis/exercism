data Candidate = Candidate {
   value :: Integer
  ,isPrime :: Bool
  ,multiplesChecked :: Bool
} deriving Show

mkCandidates num = Map.fromList lst
  where lst = map (\x -> (x,Candidate {
                                        value = x
                                      , isPrime = x == 2
                                      , multiplesChecked = x == 2
                                      })) [2..(2*num)]
